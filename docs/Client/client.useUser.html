      Returns the logged-in user's profile inside a React component.

      Prerequisite: Auth0Provider must wrap the component tree.

      Returns {
          error: Error | null | undefined;
          invalidate: () => void;
          isLoading: boolean;
          user: User | null | undefined;
      }

      { user, isLoading, error, invalidate }. user is null when logged out and undefined while loading. Call invalidate() to force a fresh profile fetch.

      Sample user object:

      {
        "sub": "auth0|64b2e5f8a3c1d90012ef4567",
        "name": "Jane Doe",
        "nickname": "jane",
        "given_name": "Jane",
        "family_name": "Doe",
        "picture": "https://cdn.auth0.com/avatars/jd.png",
        "email": "jane@example.com",
        "email_verified": true,
        "org_id": "org_abc123"
      }

      Example

      // components/profile.tsx
      import { useUser } from '@auth0/nextjs-auth0/client';

      export default function Profile() {
        const { user, isLoading, error } = useUser();

        if (isLoading) return <p>Loading...</p>;
        if (error)     return <p>Error: {error.message}</p>;
        if (!user)     return <a href="/auth/login">Log in</a>;

        return <p>Welcome, {user.name}</p>;
        //                      ^ "Jane Doe"
      }
